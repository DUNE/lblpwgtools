130  template <typename T, typename U,
131  typename std::enable_if_t<std::is_convertible_v<double, T> && std::is_convertible_v<double, U>, int> = 0>
132  decltype(T(0) - U(0)) LogLikelihood(T exp, U obs)
133  {
134  // http://www.wolframalpha.com/input/?i=d%2Fds+m*(1%2Bs)+-d+%2B+d*ln(d%2F(m*(1%2Bs)))%2Bs%5E2%2FS%5E2%3D0
135  // http://www.wolframalpha.com/input/?i=solve+-d%2F(s%2B1)%2Bm%2B2*s%2FS%5E2%3D0+for+s
136  const auto S = LLPerBinFracSystErr::GetError();
137  if(S > 0){
138  const auto S2 = util::sqr(S);
139  const auto s = .25*(sqrt(8*obs*S2+util::sqr(exp*S2-2))-exp*S2-2);
140  exp *= 1+s;
141  }
142 
143  // With this value, negative expected events and one observed
144  // event gives a chisq from this one bin of 182.
145  const double minexp = 1e-40; // Don't let expectation go lower than this
146 
147  if(obs*1000 > exp){
148  // This strange form is for numerical stability when exp ~ obs
149  return 2*obs*((exp-obs)/obs + log1p((obs-exp)/exp));
150  }
151  else{
152  // But log1p doesn't like arguments near -1 (observation much smaller
153  // than expectation), and it's better to use the usual formula in that
154  // case.
155  if(obs){
156  return 2*(exp-obs + obs*log(obs/exp));
157  }
158  else{
159  return 2*exp;
160  }
161  }
162  }
